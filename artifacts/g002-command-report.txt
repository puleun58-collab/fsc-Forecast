G002 verification commands
==========================

1. npx tsc --noEmit
2. npm run fetch:opinet
3. npm run fetch:opinet  # repeat run to verify same-date JSON update without duplication
4. set -a && . ./.env.example && set +a && npx prisma validate
5. set -a && . ./.env.example && set +a && npx prisma generate
6. set -a && . ./.env.example && set +a && npx prisma migrate diff --from-empty --to-schema-datamodel prisma/schema.prisma --script > prisma/migrations/20260706_init/migration.sql
7. Start embedded Postgres locally via embedded-postgres (verification harness only)
8. set -a && . ./.env.example && . ./.env.local && set +a && npx prisma db push --skip-generate
9. set -a && . ./.env.example && . ./.env.local && set +a && RUNTIME_ROLE=worker npm run worker
10. Repeat worker run against the same live DB to verify unchanged same-date handling
11. set -a && . ./.env.example && set +a && node --import tsx -e "...runIndicatorSync..."  # create and update indicator rows
12. set -a && . ./.env.example && . ./.env.local && set +a && node --import tsx -e "...count ingestRuns/revisions/currentRows/snapshots..."
